    private static int quantityExponent(Double val) {
        if (val == 0) {
            return 0;
        }

        int exp = (int) Math.floor(Math.log(val) / Math.log(10));

        if (val / Math.pow(10, exp) >= 10) {
            exp++;
        }

        return exp;
    }

    private static Double topDigit(Double val){
